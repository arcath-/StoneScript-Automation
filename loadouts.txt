// Place this function at the default case
// of any switch-case
func healwalk()

  // Pick up materials quickly with Star stone
  ?pickup.distance < 10
    equipL star
    equipR shield *7

  // Heal and gain armor while just walking
  // towards next objective. If 100% healed,
  // walk faster.
  :?foe.distance > 22
    ?hp < maxhp
      equipL ouroboros
      equipR shield *7
    :
      equipL triskelion
      equipR shield *7

//melee functions
func equipMeleePoison()
  equipL sword *7 poison
  equipR shield *7 poison

//Staff functions
func equipStaffPoison()
  equip staff *10 poison
  //equip ghost slayer

//armor piercing functions
func equipArmorPiercingPoison()
  equipL warhammer *10 poison +5
  equipR warhammer *10 +8

//Dashing
func equipDashing()
  equipL sword *10
  equipR shield *10 dashing

//Ranged functions
func equipRangedHealing()
  equipL crossbow *10
  equipR ouroboros

//equip the ghost slayer to regen in combat
func staffHeal()
  equip ghost slayer
